% ----------------- TERMINI CON ACRONIMO + BREVE DEFINIZIONE -------------------------
%ERP
\newacronym[description={\textit{Enterprise Resource Planning}. Software gestionale che integra tutti i processi di business (vendite, contabilità, HR) in un unico sistema informativo}]
    {erp}{ERP}{Enterprise Resource Planning}

% CRM
\newacronym [description={\textit{Software Assurance Maturity Model}. Strategia e tecnologia utilizzata per gestire le interazioni e le relazioni con i clienti attuali e potenziali, con l'obiettivo di migliorare la fidelizzazione e le vendite}]
    {crm}{CRM}{Customer Relationship Management}

% GDPR
\newacronym[description={\textit{General Data Protection Regulation}. Regolamento generale sulla protezione dei dati personali emanato dall'Unione Europea (Regolamento UE 2016/679). Stabilisce le norme relative alla raccolta, al trattamento, alla conservazione e alla cancellazione dei dati personali, imponendo alle organizzazioni precisi obblighi in materia di trasparenza, consenso e sicurezza dei dati}]
    {gdpr}{GDPR}{General Data Protection Regulation}

% PII
\newacronym[description={\textit{Personally Identifiable Information}. Qualsiasi dato che consenta di identificare direttamente o indirettamente una persona fisica (es. nome, cognome, indirizzo e-mail, numero di telefono, codice fiscale). Nel contesto del Data Catalog, i dati PII richiedono una gestione particolare in conformità alle normative vigenti}]
    {pii}{PII}{Personally Identifiable Information}

% WSL
\newacronym[description={\textit{Windows Subsystem for Linux}. Livello di compatibilità sviluppato da Microsoft che permette di eseguire un ambiente GNU/Linux nativo direttamente in Windows, senza ricorrere a macchine virtuali tradizionali o configurazioni \emph{dual-boot}}]
    {wsl}{WSL}{Windows Subsystem for Linux}

% SAMM
\newacronym[description={\textit{Software Assurance Maturity Model}. Framework progettato per aiutare le organizzazioni a formulare, implementare e misurare una strategia su misura per la sicurezza all'interno del ciclo di vita dello sviluppo del software}]
    {samm}{SAMM}{Software Assurance Maturity Model}

% NLP
\newacronym[description={\textit{Natural Language Processing}. Branca dell'Intelligenza Artificiale che si occupa dell'interazione tra computer e linguaggio umano. Nel contesto dei Data Catalog, viene utilizzata per l'analisi testuale di metadati, descrizioni e campioni di dati, al fine di estrarre significato, classificare i dataset o generare documentazione automatica}]
    {nlp}{NLP}{Natural Language Processing}

% NER
\newacronym[description={\textit{Named Entity Recognition}. Tecnica di \emph{Natural Language Processing} che identifica e classifica automaticamente le entità nominate presenti in un testo all'interno di categorie predefinite (es. nomi di persone, organizzazioni, indirizzi, numeri di telefono). È la tecnologia core utilizzata per il rilevamento automatico dei dati sensibili (PII)}]
    {ner}{NER}{Named Entity Recognition}

% SDLC
\newacronym[description={\textit{Software Development Life Cycle}. Processo strutturato utilizzato dall'industria del software per progettare, sviluppare, testare e distribuire prodotti di alta qualità. Include diverse fasi, dall'analisi dei requisiti alla manutenzione post-rilascio, garantendo il controllo, la sicurezza e la conformità di ogni fase dello sviluppo}]
    {sdlc}{SDLC}{Software Development Life Cycle}

% MCP (Server)
\newacronym[description={\textit{Model Context Protocol}. Standard aperto che normalizza la comunicazione tra i modelli linguistici di grandi dimensioni (LLM) e i sistemi informativi locali. Funge da livello semantico sicuro, permettendo agli assistenti AI di accedere al contesto dei metadati aziendali per rispondere alle interrogazioni senza esporre i dati originali all'esterno}]
    {mcp}{MCP}{Model Context Protocol}


% -------------------- VOCI DI GLOSSARIO ----------------------------
\newglossaryentry{systemintegrator}{
    name={System Integrator},
    sort=systemintegrator,
    description={Azienda che progetta e realizza infrastrutture IT complesse integrando componenti software e hardware provenienti da diversi fornitori per rispondere alle specifiche esigenze del cliente}
}

\newglossaryentry{businessintelligence}{
    name={Business Intelligence},
    sort=businessintelligence,
    description={Insieme di metodologie, architetture e strumenti che trasformano dati grezzi provenienti da fonti aziendali eterogenee in informazioni strategiche e azionabili. Il processo include l'estrazione, l'integrazione, l'analisi e la visualizzazione dei dati, consentendo al management di supportare i processi decisionali basati su evidenze oggettive e analisi predittive}
}

\newglossaryentry{anonimizzazione}{
    name={Anonimizzazione},
    sort=anonimizzazione,
    description={Processo irreversibile mediante il quale i dati personali vengono alterati in modo tale da non consentire l'identificazione, diretta o indiretta, della persona fisica a cui si riferiscono. Una volta anonimizzati, i dati non sono più soggetti alle disposizioni del GDPR}
}

\newglossaryentry{pseudonimizzazione}{
    name={Pseudonimizzazione},
    sort=pseudonimizzazione,
    description={Processo mediante il quale i dati personali vengono trattati in modo tale da non poter essere attribuiti a una persona fisica specifica senza l'utilizzo di informazioni aggiuntive. A differenza dell'anonimizzazione, la pseudonimizzazione è un processo reversibile e i dati rimangono soggetti al GDPR}
}

\newglossaryentry{dataquality}{
    name={Data Quality},
    sort=dataquality,
    description={Misura che valuta quanto un insieme di dati sia «sano» e adatto allo scopo per cui deve essere utilizzato dall'azienda. Applicare la Data Quality significa adottare un sistema che implementa controlli automatici sui dati per individuare anomalie, dati incompleti o informazioni incoerenti}
}

\newglossaryentry{datagovernance}{
    name={Data Governance},
    sort=datagovernance,
    description={Insieme di policy, ruoli, regole e processi aziendali che garantiscono che i dati siano sicuri, affidabili, accessibili e conformi alle normative. Rappresenta il «regolamento aziendale» che gestisce il ciclo di vita del dato}
}

\newglossaryentry{dataingestion}{
    name={Data Ingestion},
    sort=dataingestion,
    description={Processo di estrazione e importazione automatica dei metadati dalle fonti dati collegate al catalogo. Include la raccolta di informazioni strutturali (tabelle, colonne) ed eventuale profilazione e classificazione dei dati}
}

\newglossaryentry{datalineage}{
    name={Data Lineage},
    sort=datalineage,
    description={Rappresenta la tracciabilità del dato lungo il suo intero ciclo di vita: indica l'origine del dato, le trasformazioni che ha subito e i sistemi attraverso cui è transitato prima di giungere alla destinazione finale, aumentando la fiducia nell'affidabilità delle informazioni}
}

\newglossaryentry{dataobservability}{
    name={Data Observability},
    sort=dataobservability,
    description={Capacità di conoscere in tempo reale lo «stato di salute» dei dati. Applica i concetti di monitoraggio del software ai flussi di dati, permettendo di identificare, diagnosticare e risolvere i problemi prima che impattino gli utenti aziendali}
}

\newglossaryentry{datadiscovery}{
    name={Data Discovery},
    sort=datadiscovery,
    description={Processo di ricerca e centralizzazione volto a scoprire quali informazioni sono presenti all'interno dei vari sistemi dati aziendali. In una piattaforma di Data Catalog, agisce come un motore di ricerca intuitivo per esplorare i dataset disponibili}
}

\newglossaryentry{kubernetes}{
    name={Kubernetes},
    sort=kubernetes,
    description={Sistema \emph{open-source} per l'orchestrazione e la gestione automatizzata di \emph{container} (come Docker). Permette di automatizzare il deployment, la scalabilità e le operazioni delle applicazioni containerizzate su \emph{cluster} di server. Sebbene sia lo standard per architetture \emph{enterprise} complesse, la sua installazione e manutenzione risultano spesso troppo onerose per ambienti di collaudo o infrastrutture \emph{on-premise} di dimensioni contenute}
}

\newglossaryentry{datacontract}{
    name={Data Contract},
    sort=datacontract,
    description={Accordo formale e leggibile dalle macchine tra chi produce un dato (\emph{data producer}) e chi lo consuma (\emph{data consumer}), che definisce esplicitamente aspettative su schema, qualità, \emph{SLA} (livelli di servizio) e classificazione dei dati. Funge da contratto di garanzia per assicurare che le variazioni nei dati non interrompano i processi a valle}
}

\newglossaryentry{autoprovisioning}{
    name={Auto-Provisioning},
    sort=autoprovisioning,
    description={Processo automatizzato di configurazione e allocazione di risorse, permessi o utenze in un sistema. Nel Data Catalog, si riferisce alla capacità della piattaforma di creare automaticamente utenti, ruoli o connessioni basandosi su configurazioni predefinite o eventi scatenati dal sistema, riducendo l'intervento manuale dell'amministratore}
}

\newglossaryentry{versionamento}{
    name={Versionamento},
    sort=versionamento,
    description={Pratica di registrazione e tracciamento delle modifiche apportate a un asset (metadato, dataset o schema) nel tempo. Consente di mantenere uno storico delle evoluzioni, di confrontare diverse versioni dello stesso oggetto e di ripristinare stati precedenti in caso di errori o cambiamenti indesiderati}
}

\newglossaryentry{endtoendlineage}{
    name={End-to-End Lineage},
    sort=endtoendlineage,
    description={Visualizzazione completa del percorso di un dato dalla sorgente originale fino alla destinazione finale (es. report o dashboard). Consente di tracciare le trasformazioni, le pipeline e i sistemi intermedi coinvolti nel flusso, fornendo una visione globale della circolazione dei dati nell'infrastruttura}
}

\newglossaryentry{dataobjectlineage}{
    name={Data Object Lineage},
    sort=dataobjectlineage,
    description={Tracciabilità del percorso di un oggetto dati specifico (tavola, vista, file) tra i sistemi. A differenza del \emph{Column-Level Lineage}, si concentra sulle dipendenze tra entità di alto livello, tralasciando il dettaglio dei singoli campi}
}

\newglossaryentry{columnlevellineage}{
    name={Column-Level Lineage},
    sort=columnlevellineage,
    description={Livello più granulare di tracciabilità, che identifica come una specifica colonna di un dataset di destinazione sia stata derivata da una o più colonne sorgente attraverso trasformazioni logiche o calcoli. È essenziale per l'analisi di impatto e per tracciare la propagazione di dati sensibili}
}

\newglossaryentry{productbacklog}{
    name={Product Backlog},
    sort=productbacklog,
    description={Lista ordinata e dinamica di tutte le funzionalità, requisiti, modifiche e correzioni necessarie per il prodotto. Rappresenta la fonte unica di lavoro per il team di sviluppo nella metodologia Scrum, in cui gli elementi vengono selezionati e prioritizzati per l'inserimento negli \emph{Sprint}}
}

\newglossaryentry{datamesh}{
    name={Data Mesh},
    sort=datamesh,
    description={Approccio architettonico e organizzativo alla gestione dei dati che supera il modello centralizzato del Data Warehouse/Data Lake. Si basa su quattro pilastri: proprietà decentralizzata dei dati (orientata ai domini), dati come prodotto (\emph{data as a product}), piattaforma di dati self-service e governance federata. Promuove l'autonomia dei team di business nella gestione dei propri dati}
}

\newglossaryentry{embedding}{
    name={embedding vettoriali},
    sort=embedding,
    description={Rappresentazione matematica di parole, frasi o concetti sotto forma di vettori numerici ad alta dimensionalità. Nel contesto della ricerca semantica, gli \emph{embedding} permettono al motore di ricerca di comprendere il significato e il contesto di una query espressa in linguaggio naturale, superando i limiti della semplice corrispondenza di parole chiave}
}

\newglossaryentry{opensearch}{
    name={OpenSearch},
    sort=opensearch,
    description={Motore di ricerca e analisi distribuito \emph{open-source} (derivato da ElasticSearch). Nel contesto di OpenMetadata, viene utilizzato come \emph{backend} per alimentare le funzionalità di ricerca \emph{full-text} e la ricerca semantica, indicizzando tutti i metadati presenti nel catalogo per consentire interrogazioni rapide su grandi volumi di informazioni}
}
